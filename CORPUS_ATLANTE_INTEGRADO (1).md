import numpy as np
from scipy.integrate import odeint

def kuramoto_toroidal(theta, t, omega, K, A, alpha):

# Cargar mapa CMB de Planck
cmb_map = hp.read_map('COM_CMB_IQU-smica_2048_R3.00_full.fits')

# Calcular espectro de potencia angular
cl = hp.anafast(cmb_map)
ell = np.arange(len(cl))

# Función de correlación
theta = np.linspace(0, np.pi, 1000)
C_theta = np.zeros_like(theta)
for i, t in enumerate(theta):
    C_theta[i] = np.sum((2*ell+1)/(4*np.pi) * cl * hp.sphtfunc.legendre(ell, np.cos(t)))

R_CMB = params[0]
